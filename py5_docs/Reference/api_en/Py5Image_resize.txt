@@ example
image = Py5Image_resize_0.png

def setup():
    py5.rect(10, 10, 80, 80)
    py5.fill(255, 0, 0)
    py5.rect(15, 15, 10, 60)
    img = py5.get_pixels()
    for x in [80, 60, 40]:
        img.resize(x, x, py5.BICUBIC)
        py5.image(img, 100 - x, 100 - x)
