@@ meta
name = is3d()
type = method
pclass = PGraphics
processing_name = is3D

@@ signatures
is3d() -> bool

@@ description
Boolean value reflecting if the Py5Graphics object is or is not a 3D renderer.

This will be `True` if the renderer is a 3D renderer such as `P3D`.

@@ example
def setup():
    py5.size(200, 200)
    g = py5.get_graphics()

    # prints False
    print(g.is3d())

@@ example
def setup():
    py5.size(200, 200, py5.P3D)
    g = py5.get_graphics()

    # prints True
    print(g.is3d())
